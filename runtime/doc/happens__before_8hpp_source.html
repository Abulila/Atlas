29 // This structure is currently used in a write-once manner. It contains
30 // the core information within an entry in a hash table. If any of the

44 
45  // Here LogAddr_ is the address of the log entry of the last release
46  // operation of the corresponding synchronization object.
47  LogEntry *LogAddr;
48 
49  // The following map contains the set of locks (and their counts)
50  // that *this* lock *depends* on. This *dependence* relation is
51  // established at the point *this* lock is released. At any point
52  // of time, there can be at most one writer since only one thread
53  // can release *this* lock. But there can be multiple readers who
54  // may be examining this map (e.g. rw-locks).
55  MapOfLockInfo *LockInfoPtr;
