230 inline void *PRegionMgr::computePRegionBaseAddr(void *addr) const
231 {
232  assert(addr >= (static_cast<char*>(PRegionTable_) + kPRegionSize_) &&
233  addr < (static_cast<char*>(PRegionTable_) +
234  kPRegionSize_ * (kMaxNumPRegions_ + 1)) &&
235  "Location not in any persistent region!");
236  return static_cast<char*>(PRegionTable_) + kPRegionSize_ *
237  ((reinterpret_cast<uint64_t>(addr) -
238  reinterpret_cast<uint64_t>(PRegionTable_))/kPRegionSize_);
239 }

244 inline void *PRegionMgr::computeNewPRegionBaseAddr() const
245 {
246  uint32_t num_regions = getNumPRegions();
247  if (!num_regions) return (char*)kPRegionsBase_ + kPRegionSize_;
248  else return static_cast<char*>(
249  reinterpret_cast<PRegion*>(
250  reinterpret_cast<char*>(getPRegionArrayPtr()) +
251  (num_regions - 1) * sizeof(PRegion))->
252  get_base_addr()) + kPRegionSize_;
